import Foundation
func main(argc: Int32, argv: Int8**) -> Int32
{
	var foo: Bar
	printf("%d\n", foo.volume())
}

struct Bar
{
	var Length: Int32 = 17
	var Width: Int32 = 5
	var Thickness: Int32 = 2

	func volume() -> Int32
	{
		return self.Length * self.Width * self.Thickness
	}
}
